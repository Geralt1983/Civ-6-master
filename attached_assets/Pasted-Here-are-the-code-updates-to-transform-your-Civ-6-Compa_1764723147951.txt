This stops the "False Positive" saves by waiting for Civ 6 to finish writing its log block before sending.

        hostname: 'civ-sage-ai--jkimble1983.replit.app',

        method: 'POST',
        headers: {
            'Content-Type': 'application/json',

        }
    };
    
    const req = https.request(options, (res) => {
        if (res.statusCode === 200) console.log('[SUCCESS] Updated');
        else console.log(`[ERROR] Server: ${res.statusCode}`);
    });
    
    req.on('error', (e) => console.error('[ERROR]', e.message));
    req.write(postData);
    req.end();
};

tail.on('line', (line) => {
    if (line.includes('CIV_DATA_DUMP::')) {
        const startIdx = line.indexOf('{');
        const endIdx = line.lastIndexOf('}') + 1;
        
        if (startIdx !== -1 && endIdx > startIdx) {
            const jsonStr = line.substring(startIdx, endIdx);
            try {
                // Parse and debounce
                const gameData = eval('(' + jsonStr + ')');
                lastGameData = gameData;
                
                // Clear previous timer if exists
                if (debounceTimer) clearTimeout(debounceTimer);
                
                // Wait 2 seconds of silence before sending
                debounceTimer = setTimeout(() => {
                    sendData(lastGameData);
                }, 2000);
                
                console.log('[DETECTED] Data buffered...');
            } catch (e) {
                // ignore parse errors
            }
        }
    }

4. Architecture: Local Persistence
This ensures you don't lose your data if you refresh the browser or the Replit server sleeps.
File: client/src/hooks/useGameState.ts
import { useQuery, useQueryClient } from "@tanstack/react-query";
import { useEffect, useRef } from "react";
import type { GameState } from "@/lib/mockData";

export function useGameState() {
  const queryClient = useQueryClient();
  const wsRef = useRef<WebSocket | null>(null);

  // Restore from LocalStorage on load
  useEffect(() => {
    const saved = localStorage.getItem("civ6_gamestate");
    if (saved) {
      try {
        const parsed = JSON.parse(saved);
        queryClient.setQueryData(["gamestate"], parsed);
      } catch (e) { console.error("Failed to load cached state"); }
    }
  }, [queryClient]);

  const { data: gameState, isLoading } = useQuery<GameState>({
    queryKey: ["gamestate"],
    queryFn: async () => {
      const response = await fetch("/api/gamestate");
      if (!response.ok) return null;
      const data = await response.json();
      // Cache valid server response
      localStorage.setItem("civ6_gamestate", JSON.stringify(data));
      return data;
    },
    refetchInterval: 5000, 
  });

  useEffect(() => {
    const protocol = window.location.protocol === "https:" ? "wss:" : "ws:";
    const wsUrl = `${protocol}//${window.location.host}/ws`;
    const ws = new WebSocket(wsUrl);
    wsRef.current = ws;

    ws.onmessage = (event) => {
      try {
        const message = JSON.parse(event.data);
        if (message.type === "gamestate") {
          queryClient.setQueryData(["gamestate"], message.data);
          // Cache WebSocket updates
          localStorage.setItem("civ6_gamestate", JSON.stringify(message.data));
        }
      } catch (error) {
        console.error("Error parsing WebSocket message:", error);
      }
    };

    return () => ws.close();
  }, [queryClient]);

  return { gameState, isLoading };
}
